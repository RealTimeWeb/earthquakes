001    package realtimeweb.earthquakeservice.domain;
002    
003    
004    import com.google.gson.Gson;
005    import com.google.gson.GsonBuilder;
006    import com.google.gson.JsonArray;
007    import com.google.gson.JsonElement;
008    import com.google.gson.JsonObject;
009    import com.google.gson.JsonParser;
010    
011    import java.util.ArrayList;
012    import java.util.Arrays;
013    import java.util.HashMap;
014    import java.util.List;
015    import java.util.Arrays;
016    
017    /**
018     * Information about earthquakes matching certain criteria, including the area that they occurred.
019     */
020    public class Report {
021            
022            
023            private BoundingBox area;
024            private ArrayList<Earthquake> earthquakes;
025            private String title;
026            
027            
028            /**
029             * A region that contains all the earthquakes.
030            
031             * @return bounding box
032             */
033            public BoundingBox getArea() {
034                    return this.area;
035            }
036            
037            /**
038             * 
039             * @param area A region that contains all the earthquakes.
040             */
041            public void setArea(BoundingBox area) {
042                    this.area = area;
043            }
044            
045            /**
046             * A list of the earthquakes.
047            
048             * @return ArrayList<Earthquake>
049             */
050            public ArrayList<Earthquake> getEarthquakes() {
051                    return this.earthquakes;
052            }
053            
054            /**
055             * 
056             * @param earthquakes A list of the earthquakes.
057             */
058            public void setEarthquakes(ArrayList<Earthquake> earthquakes) {
059                    this.earthquakes = earthquakes;
060            }
061            
062            /**
063             * A human-readable title that describes this data.
064            
065             * @return String
066             */
067            public String getTitle() {
068                    return this.title;
069            }
070            
071            /**
072             * 
073             * @param title A human-readable title that describes this data.
074             */
075            public void setTitle(String title) {
076                    this.title = title;
077            }
078            
079            
080            
081            /**
082             * Information about earthquakes matching certain criteria, including the area that they occurred.
083            
084             * @return String
085             */
086            public String toString() {
087                    return "Report[" + area + ", " + earthquakes + ", " + title + "]";
088            }
089            
090            /**
091             * Internal constructor to create a Report from a Json representation.
092             * @param json The raw json data that will be parsed.
093             * @param gson The Gson parser. See <a href='https://code.google.com/p/google-gson/'>https://code.google.com/p/google-gson/</a> for more information.
094             * @return 
095             */
096            public  Report(JsonObject json, Gson gson) {
097                    this.area = new BoundingBox(json.get("bbox").getAsJsonArray(), gson);
098                    this.earthquakes = new ArrayList<Earthquake>();
099                    for (JsonElement e : json.get("features").getAsJsonArray()) {
100                            this.earthquakes.add(new Earthquake(e.getAsJsonObject(), gson));
101                    }
102                    this.title = json.get("metadata").getAsJsonObject().get("title").getAsString();
103            }
104            
105            /**
106             * Regular constructor to create a Report.
107             * @param area A region that contains all the earthquakes.
108             * @param earthquakes A list of the earthquakes.
109             * @param title A human-readable title that describes this data.
110             * @return 
111             */
112            public  Report(BoundingBox area, ArrayList<Earthquake> earthquakes, String title) {
113                    this.area = area;
114                    this.earthquakes = earthquakes;
115                    this.title = title;
116            }
117            
118    }

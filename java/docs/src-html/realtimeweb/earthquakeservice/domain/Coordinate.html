013    /**
014     * The longitudinal, latitudinal, and depth where the earthquake occurred.
015     */
016    public class Coordinate {
017            
018        // For some unclear reason, these are stored as a list instead of a dictionary.
019        
020        private Double longitude;
021        private Double latitude;
022        private Double depth;
023        
024        
025        /**
026         * @return The longitude (West-North) component.
027         */
028        public Double getLongitude() {
029            return this.longitude;
030        }
031        
032        /**
033         * @param The longitude (West-North) component.
034         * @return Double
035         */
036        public void setLongitude(Double longitude) {
037            this.longitude = longitude;
038        }
039        
040        /**
041         * @return The latitude (South-North) component.
042         */
043        public Double getLatitude() {
044            return this.latitude;
045        }
046        
047        /**
048         * @param The latitude (South-North) component.
049         * @return Double
050         */
051        public void setLatitude(Double latitude) {
052            this.latitude = latitude;
053        }
054        
055        /**
056         * @return The depth (closer or farther from the surface) component.
057         */
058        public Double getDepth() {
059            return this.depth;
060        }
061        
062        /**
063         * @param The depth (closer or farther from the surface) component.
064         * @return Double
065         */
066        public void setDepth(Double depth) {
067            this.depth = depth;
068        }
069        
070            
071            /**
072             * Creates a string based representation of this Coordinate.
073            
074             * @return String
075             */
076            public String toString() {
077                    return "Coordinate[" +longitude+", "+latitude+", "+depth+"]";
078            }
079            
080            /**
081             * Internal constructor to create a Coordinate from a json representation.
082             * @param map The raw json data that will be parsed.
083             * @return 
084             */
085        public Coordinate(List<Object> raw) {
086            // TODO: Check that the data has the correct schema.
087            // NOTE: It's much safer to check the Map for fields than to catch a runtime exception.
088            try {
089                this.longitude = Double.parseDouble(raw.get(0).toString());
090                this.latitude = Double.parseDouble(raw.get(1).toString());
091                this.depth = Double.parseDouble(raw.get(2).toString());
092            } catch (NullPointerException e) {
093                    System.err.println("Could not convert the response to a Coordinate; a field was missing.");
094                    e.printStackTrace();
095            } catch (ClassCastException e) {
096                    System.err.println("Could not convert the response to a Coordinate; a field had the wrong structure.");
097                    e.printStackTrace();
098            }
099        
100            }       
101    }
